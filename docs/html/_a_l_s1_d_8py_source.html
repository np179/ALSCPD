127 def get_b_ein(V, nu_store, hole_index):
128  '''Function to iteratively create input for the np.einsum routine to contract a given tensor with given
129  set of SPP for all DOF except one indicated.
130 
131  [Args]:
132  V[array]: The exact tensor to be contracted with shape (N0, N1, N2,... Nf).
133  nu_store[list]: List of the SPP with index running over the DOF in shape (r,N).

138 
139  -tested for up to 6D potential.'''

144  # get a reference, this could be any integer but cannot be lower than the DOF
145  ref = np.ndim(V)

149  # iteration it will be N->N, every following one will be N->N-1
150  next_dim = ref

152  old_shape = np.arange(0, ref).tolist()
153  # go through all the nu
154  for nu in nu_store:

156  if i != hole_index:
157  # if no, check if we are in the last iteration
158  if i > 0:
